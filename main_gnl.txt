int	main(void)
{
	int		i;
	int		file;
	char	*line;
	i = 0;
	file = open("file.txt", O_RDONLY);
	if (file == -1)
	{
		printf("Fail to open file\n");
		return (1);
	}
	while (++i <= 8)
	{
		line = get_next_line(file);
		if (!line)
			printf("Call %d = (null)\n", i);
		else
			printf("Call %d = %s", i, (char *)line);
		free(line);
		line = NULL;
	}
	close(file);
	return (0);
}
